Blinking LED:
import RPi.GPIO as GPIO
from time import sleep
GPIO.setwarnings(False)
GPIO.setmode(GPIO.BOARD)
GPIO.setup(8, GPIO.OUT, initial=GPIO.LOW)
while True:
GPIO.output(8, GPIO.HIGH)
sleep(1)
GPIO.output(8, GPIO.LOW)
sleep(1)

Traffic Light:
from gpiozero import LED
red = LED(22)
amber = LED(27)
green = LED(17)
red.blink(1, 1)
amber.blink(2, 2)
green.blink(3, 3)
